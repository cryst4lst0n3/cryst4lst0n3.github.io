```c
__int64 check()
{
  unsigned __int8 v1; // al
  unsigned __int8 chr; // [rsp+Fh] [rbp-11h]
  _QWORD flag[2]; // [rsp+10h] [rbp-10h]

  flag[1] = __readfsqword(0x28u);
  flag[0] = 0;
  while ( 1 )
  {
    chr = inpt[ptr_inpt++];
    if ( chr == 32 )
      break;
    switch ( chr )
    {
      case '!':
        putchar(inpt[ptr_inpt]);
        ++ptr_inpt;
        break;
      case '"':
        v1 = getchar();
        sub_12DC(v1);
        break;
      case '#':
        *((_BYTE *)flag + inpt[ptr_inpt]) = inpt[ptr_inpt + 1];
        ptr_inpt += 2;
        break;
      case '$':
        *((_BYTE *)flag + inpt[ptr_inpt]) = *((_BYTE *)flag + inpt[ptr_inpt + 1]);
        ptr_inpt += 2;
        break;
      case '%':
        *((_BYTE *)flag + inpt[ptr_inpt]) = *((_BYTE *)flag + inpt[ptr_inpt + 2])
                                          & *((_BYTE *)flag + inpt[ptr_inpt + 1]);
        ptr_inpt += 3;
        break;
      case '&':
        *((_BYTE *)flag + inpt[ptr_inpt]) = *((_BYTE *)flag + inpt[ptr_inpt + 2])
                                          | *((_BYTE *)flag + inpt[ptr_inpt + 1]);
        ptr_inpt += 3;
        break;
      case '\'':
        *((_BYTE *)flag + inpt[ptr_inpt]) = *((_BYTE *)flag + inpt[ptr_inpt + 2])
                                          ^ *((_BYTE *)flag + inpt[ptr_inpt + 1]);
        ptr_inpt += 3;
        break;
      case '(':
        LOBYTE(flag[0]) = sub_1315();
        break;
      case ')':
        sub_12DC(*((unsigned __int8 *)flag + inpt[ptr_inpt]));
        ++ptr_inpt;
        break;
    }
  }
  return LOBYTE(flag[0]);
}
```
